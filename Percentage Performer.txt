    <script>
        function addSemesterFields() {
            var numSemesters = parseInt(document.getElementById('numSemesters').value);
            var semesterInputsDiv = document.getElementById('semesterInputs');
            semesterInputsDiv.innerHTML = '';

            for (var i = 1; i <= numSemesters; i++) {
                semesterInputsDiv.innerHTML += "<h3>Semester " + i + ":</h3>";

                var numSubjects = prompt("How many subjects in Semester " + i + "?");
                semesterInputsDiv.innerHTML += "<input type='hidden' id='numSubjects" + i + "' value='" + numSubjects + "'>";

                for (var j = 1; j <= numSubjects; j++) {
                    semesterInputsDiv.innerHTML += "<label for='semester" + i + "subject" + j + "'>Subject " + j + ":</label>";
                    semesterInputsDiv.innerHTML += "<input type='number' id='semester" + i + "subject" + j + "' name='semester" + i + "subject" + j + "' required>";
                    
                }
            }
        }

        function calculateTotal() {
            var numSemesters = parseInt(document.getElementById('numSemesters').value);
            var totalMarks = 0;
            var totalSubjects = 0;
            var totalGradePoints = 0;

            for (var i = 1; i <= numSemesters; i++) {
                var numSubjects = parseInt(document.getElementById('numSubjects' + i).value);
                totalSubjects += numSubjects;

                for (var j = 1; j <= numSubjects; j++) {
                    var subjectMarks = parseFloat(document.getElementById('semester' + i + 'subject' + j).value);
                    totalMarks += subjectMarks;

                    
                }
            }

            var percentage = (totalMarks / (totalSubjects * 100)) * 100;
            var cgpa = percentage / 9.8;

            document.getElementById('totalMarks').innerHTML = "<h3>Total Marks: " + totalMarks + "</h3>";
            document.getElementById('percentage').innerHTML = "<h3>Percentage: " + percentage.toFixed(2) + "%</h3>";
            document.getElementById('cgpa').innerHTML = "<h3>CGPA: " + cgpa.toFixed(2) + "</h3>";
        }
    </script>
